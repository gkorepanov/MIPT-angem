\def \fixedr {
    \item Даны прямая $\mathbf{r} = \mathbf{r_0} + \mathbf{a}t$ и плоскость $(\mathbf{r}, \mathbf{n})=D$, не параллельные между собой. Точка $M$ лежит на прямой и удалена от плоскости на расстояние $\rho$. Найти радиус-вектор точки $M$.
}


\def \four {
	\item Найдите радиус-вектор точки пересечения прямой $[\mathbf{r}, \mathbf{a}]=\mathbf{b}$ и плоскости ${(\mathbf{r}, \mathbf{n})=D}$, если $(\mathbf{a}, \mathbf{n}) \neq 0$
}



\def \rasta {
	\item Найдите расстояние от точки $M_0(\vec{r_0})$ до прямой $[\vec{r}, \vec a] = \vec b$
}

\def \rastb {
	\item Найдите расстояние между параллельными плоскостями $(\vec r, \vec n) = D_1$ и ${(\vec r, \vec n) = D_2}$.
}
